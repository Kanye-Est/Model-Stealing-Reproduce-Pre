\begin{frame}{Equation-Solving: Confidence Values Become Equations}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.50\textwidth}
      \begin{block}{Binary Logistic Regression}
        \[
          p = \sigma(w^\top x + b), \qquad
          \log\frac{p}{1-p} = w^\top x + b
        \]
        \begin{itemize}
          \item Breast Cancer has $d=30$ features.
          \item Unknowns are $w_1,\dots,w_{30},b$.
          \item $d+1=31$ independent confidence queries give a linear system.
          \item Solving $[X,1]\theta=\mathrm{logit}(p)$ recovers the model.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.46\textwidth}
      \centering
      \begin{tikzpicture}[
        node distance=0.58cm,
        box/.style={draw=linegray, rounded corners, align=center, fill=softgray, minimum width=3.35cm, minimum height=0.62cm},
        arr/.style={-{Latex[length=2mm]}, thick, color=titleblue}
      ]
        \node[box] (q) {queries $x_i$};
        \node[box, below=of q] (p) {confidence $p_i=f(x_i)$};
        \node[box, below=of p] (eq) {$\mathrm{logit}(p_i)$};
        \node[box, below=of eq] (solve) {linear solve / optimization};
        \node[box, below=of solve] (clone) {clone $f'$};
        \draw[arr] (q) -- (p);
        \draw[arr] (p) -- (eq);
        \draw[arr] (eq) -- (solve);
        \draw[arr] (solve) -- (clone);
      \end{tikzpicture}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Decision Trees: Differential Testing with Leaf IDs}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.46\textwidth}
      \centering
      \includegraphics[width=\textwidth]{author_slides/tree_author-14.png}
      \source{Source: Tram\`er et al., USENIX Security 2016 slides, p.14}
    \end{column}
    \begin{column}{0.50\textwidth}
      \begin{block}{Path-finding in the reproduction}
        \begin{enumerate}
          \item Query a point and record its leaf id.
          \item Change one feature at a time.
          \item A changed leaf id indicates a split on that feature.
          \item Use binary search for continuous thresholds.
          \item Recurse on newly discovered regions.
        \end{enumerate}
      \end{block}
      \begin{block}{Result on Iris}
        Target leaves: 5; extracted leaves: 5; queries: 557; test and uniform agreement: 100\%.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Decision Tree Implementation Detail}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.60\textwidth}
      \centering
      \includegraphics[width=\textwidth]{tree_leaf_id_mechanism.png}
    \end{column}
    \begin{column}{0.37\textwidth}
      \begin{block}{Why the naive scan fails}
        If only the two interval endpoints are checked, both may return the same leaf id while an unseen middle region reaches a different leaf.
      \end{block}
      \begin{block}{Implementation choice}
        The extractor recovers predicates along the current path and tracks interval constraints per feature. This avoids missing hidden leaves and handles $\leq$ / $>$ threshold boundaries explicitly.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{KLR Leakage: Parameters Can Contain Training Samples}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.42\textwidth}
      \scriptsize
      \begin{block}{Representer model used for visualization}
        \[
          s_c(x)=\sum_{r:y_r=c}\exp(-\gamma\|x-r\|^2)
        \]
        Digits dataset; 5 representers per class; 50 extracted images in total.
      \end{block}
      \begin{block}{Security implication}
        Extraction may reveal more than functionality. For representer-style kernel models, recovered parameters can directly expose training examples.
      \end{block}
    \end{column}

\begin{frame}{Reproduction Takeaways}
  \begin{block}{What was reproduced locally}
    \begin{itemize}
      \item Confidence APIs make LR extraction a system-solving problem.
      \item Multiclass LR reaches perfect agreement at roughly one query per parameter.
      \item Decision-tree extraction depends on leaf identity and careful path predicates.
      \item Label-only APIs raise query cost but do not eliminate extraction.
      \item Representer-style kernel models illustrate data leakage risk.
    \end{itemize}
  \end{block}
  \begin{block}{Boundary of this reproduction}
    AWS/BigML online experiments were not rerun; the goal is a local, auditable reproduction of the attack mechanisms for presentation.
  \end{block}
\end{frame}
